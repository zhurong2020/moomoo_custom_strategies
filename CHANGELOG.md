def calculate_add_position_qty(self, drawdown):
    """每次达到阈值都加仓 - 最大化抄底机会"""
    for i, threshold in enumerate(self.drawdown_layers):
        if drawdown >= threshold:
            # 移除层级限制 - 激进抄底

    return 0
